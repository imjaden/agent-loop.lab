def run(self, task: str, verbose=True,
        checkpoint: str = "") -> str:
    self.messages = []
    self.history = []

    self.state = LoopState(                ← 初始化状态
        task_id=hashlib.md5(task.encode()).hexdigest()[:12],
    )

    if self.system_prompt:
        self.messages.append({                   ← 注入 system prompt
            "role": "system",
            "content": self.system_prompt,
        })
    self.messages.append({                       ← 注入用户任务
        "role": "user",
        "content": task,
    })

# 验证器定义
class Verifier:
    DEFAULT_CHECKS = [
        ("最小长度",  lambda t: len(t) > 100,
         "内容过短（<100 chars）"),
        ("有实质内容", lambda t: len(t.split()) > 20,
         "内容缺少实质性信息"),
    ]

    def verify(self, text: str) -> bool:
        self.failures = []
        for name, fn, msg in self.checks:
            if not fn(text):
                self.failures.append(f"  ❌ [{name}] {msg}")
        return len(self.failures) == 0
